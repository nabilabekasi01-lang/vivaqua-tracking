  <div class="legend">
    <div><span style="background: blue;"></span> Supir</div>
    <div><span style="background: red;"></span> Toko Langganan</div>
  </div>

  <script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>
  <script>
    // Ganti ID di sini
    const sheetSupir = '1AAAaaAAA111'; // ID Google Sheet supir
    const sheetToko  = '2BBBbbBBB222'; // ID Google Sheet toko
    const urlSupir = `https://opensheet.elk.sh/${sheetSupir}/Sheet1`;
    const urlToko  = `https://opensheet.elk.sh/${sheetToko}/Sheet1`;

    const map = L.map('map').setView([-6.2, 106.8], 10);
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      attribution: '&copy; OpenStreetMap contributors'
    }).addTo(map);

    let markersSupir = [];
    let markersToko = [];

    async function loadSupir() {
      const res = await fetch(urlSupir);
      const data = await res.json();
      markersSupir.forEach(m => map.removeLayer(m));
      markersSupir = [];

      data.forEach(row => {
        if (row.latitude && row.longitude) {
          const marker = L.circleMarker([parseFloat(row.latitude), parseFloat(row.longitude)], {
            color: 'blue',
            radius: 8,
            fillOpacity: 0.8
          })
          .addTo(map)
          .bindPopup(`<b>Supir: ${row.device_id}</b><br>
                      Speed: ${row.speed} km/h<br>
                      Waktu: ${row.timestamp}`);
          markersSupir.push(marker);
        }
      });
    }

    async function loadToko() {
      const res = await fetch(urlToko);
      const data = await res.json();
      markersToko.forEach(m => map.removeLayer(m));
      markersToko = [];

      data.forEach(row => {
        if (row.latitude && row.longitude) {
          const marker = L.circleMarker([parseFloat(row.latitude), parseFloat(row.longitude)], {
            color: 'red',
            radius: 7,
            fillOpacity: 0.9
          })
          .addTo(map)
          .bindPopup(`<b>Toko: ${row.nama_toko}</b>`);
          markersToko.push(marker);
        }
      });
    }

    // Muat data pertama kali
    loadSupir();
    loadToko();

    // Refresh otomatis tiap 15 detik
    setInterval(() => {
      loadSupir();
      loadToko();
    }, 15000);
  </script>
